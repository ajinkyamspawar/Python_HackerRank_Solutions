 # Enter your code here. Read input from STDIN. Print output to STDOUT
from collections import defaultdict

a, b = map(int, input().split())

# initialise defaultdict with list as as default factory 

group_a = defaultdict(list)

# Read words in group_a and store their 1-indexed positions 

for i in range (1, a+1): 
    word = input().strip()
    group_a[word].append(i)
    
#Read words in group B and ouput the corresponding indices or -1

for _ in range(b):
    word = input().strip()
    if word in group_a:
        print(" ".join(map(str, group_a[word])))
    else: 
        print(-1)
